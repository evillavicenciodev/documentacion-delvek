   //FrontGeisha/Areas/AREAFRONT/views/PREFCONTROLFRONT/Index.cshtml

    @model IEnumerable<CLASSMODELO>

    @inject IServicioSeguridad servicioSeguridad

    @{
        ViewData["Title"] = "DESCPLURAL";
        bool AccesoCrear = await servicioSeguridad.Acceso(this.ViewContext.RouteData, "Crear");
        bool AccesoEditar = await servicioSeguridad.Acceso(this.ViewContext.RouteData, "Editar");
        bool AccesoBorrar = await servicioSeguridad.Acceso(this.ViewContext.RouteData, "Borrar");
    }

    <h1>DESCPLURAL</h1>

    <div asp-validation-summary="ModelOnly" class="text-danger"></div>
    @if (AccesoCrear)
    {
        <div class="mb-3">
            <a asp-action="Crear" class="btn btn-primary" title="Crear DESCSINGULAR"><i class="ICONOCREAR me-2"></i>Crear DESCSINGULAR</a>
        </div>
    }

    <div class="card">
        <div class="card-body">

            <div class="my-scrollbar scroll-8">
                <table id="tabla" class="table table-bordered table-hover table-striped">
                    <thead class="encabezado encabezado-fijo">
                        <tr>
                            <th>Acciones</th>
                            <th>DISPLAYCAMPODESC</th>
                        </tr>
                    </thead>
                    <tbody>
                        @foreach (var VARFOREACH in Model)
                        {
                            <tr>
                                <td>
                                    <input type="hidden" value="@VARFOREACH.Id" />
                                    @if (AccesoEditar)
                                    {
                                        <a class="btn btn-primary" asp-action="Editar" asp-route-id="@VARFOREACH.Id" title="Editar"><i class="bi-pencil"></i></a>
                                    }
                                    @if (AccesoBorrar)
                                    {
                                        <button type="button" class="btn btn-danger" onclick="confirmaBorrar(this)" title="Borrar"><i class="bi-trash"></i></button>
                                    }
                                </td>
                                <td>
                                    @VARFOREACH.CAMPODESCRIP
                                </td>
                            </tr>
                        }
                    </tbody>
                </table>
            </div>

        </div>
    </div>

    @section Scripts {
        <partial name="_ValidationScriptsPartial" />
        <script src="~/js/utilidades.js" asp-append-version="true"></script>

        <script>
            function confirmaBorrar(boton) {

                const renglon = $(boton).closest("TR");
                const indice = renglon[0].rowIndex;
                const tr = $("#tabla")[0].getElementsByTagName("tr");
                const Id = parseInt(tr[indice].getElementsByTagName("input")[0].value);
                const CAMPODESCRIP = $("td", renglon).eq(1).html().trim();

                confirmarAccion({
                    callBackAceptar: () => {
                        borrarRegistro(Id, indice);
                    },
                    callbackCancelar: () => {

                    },
                    titulo: `¿Desea borrar ${CAMPODESCRIP}?`
                })
            }

            async function borrarRegistro(Id, indice) {
                const url = '@Url.Action("Borrar")/' + Id;
                const respuesta = await fetch(url, {
                    method: 'DELETE'
                });

                if (!respuesta.ok) {
                    manejarErrorApi(respuesta);
                    return;
                }

                $("#tabla")[0].deleteRow(indice);
            }

        </script>
    }
